Calculadora en Python

Una calculadora funcional desarrollada en Python con interfaz gráfica usando Tkinter. Permite realizar operaciones básicas como suma, resta, multiplicación y división.

Python Tkinter GUI
Ver código 
import tkinter as tk
from tkinter import messagebox
import math

class Calculadora:
    def __init__(self, root):
        self.ventana = root
        self.ventana.title("Calculadora Avanzada")
        self.ventana.configure(bg="#1e1e1e")
        self.expresion = ""
        self.historial = []
        self.entrada = tk.StringVar()
        self.crear_interfaz()
        self.ventana.bind("<Key>", self.tecla_presionada)

    def crear_interfaz(self):
        entrada = tk.Entry(self.ventana, textvariable=self.entrada, font=("Arial", 24), bd=10, width=22, bg="#2e2e2e", fg="white", insertbackground="white", justify="right")
        entrada.grid(row=0, column=0, columnspan=4)

        botones = [
            ('7', 1, 0), ('8', 1, 1), ('9', 1, 2), ('/', 1, 3),
            ('4', 2, 0), ('5', 2, 1), ('6', 2, 2), ('*', 2, 3),
            ('1', 3, 0), ('2', 3, 1), ('3', 3, 2), ('-', 3, 3),
            ('0', 4, 0), ('.', 4, 1), ('+', 4, 2), ('^', 4, 3),
            ('C', 5, 0), ('√', 5, 1), ('=', 5, 2), ('Hist', 5, 3)
        ]

        for (texto, fila, columna) in botones:
            comando = lambda t=texto: self.click_boton(t)
            tk.Button(self.ventana, text=texto, padx=20, pady=20, font=("Arial", 16),
                      command=comando, bg="#3a3a3a", fg="white", activebackground="#505050", activeforeground="white")\
                .grid(row=fila, column=columna, sticky="nsew")

        for i in range(6):
            self.ventana.grid_rowconfigure(i, weight=1)
            self.ventana.grid_columnconfigure(i % 4, weight=1)

    def click_boton(self, valor):
        if valor == "C":
            self.expresion = ""
            self.entrada.set("")
        elif valor == "=":
            self.evaluar()
        elif valor == "√":
            try:
                resultado = math.sqrt(eval(self.expresion))
                self.historial.append(f"√({self.expresion}) = {resultado}")
                self.expresion = str(resultado)
                self.entrada.set(self.expresion)
            except:
                self.mostrar_error()
        elif valor == "Hist":
            self.mostrar_historial()
        else:
            if valor == "^":
                valor = "**"
            self.expresion += valor
            self.entrada.set(self.expresion)

    def evaluar(self):
        try:
            resultado = eval(self.expresion)
            self.historial.append(f"{self.expresion} = {resultado}")
            self.expresion = str(resultado)
            self.entrada.set(self.expresion)
        except:
            self.mostrar_error()

    def mostrar_error(self):
        messagebox.showerror("Error", "Expresión no válida")
        self.expresion = ""
        self.entrada.set("")

    def mostrar_historial(self):
        if not self.historial:
            messagebox.showinfo("Historial", "No hay operaciones aún.")
        else:
            historial_texto = "\n".join(self.historial[-10:])
            messagebox.showinfo("Historial (últimas 10)", historial_texto)

    def tecla_presionada(self, evento):
        tecla = evento.char
        if tecla in "0123456789.+-*/()":
            self.expresion += tecla
            self.entrada.set(self.expresion)
        elif tecla == "\r":
            self.evaluar()
        elif tecla.lower() == "c":
            self.expresion = ""
            self.entrada.set("")
        elif tecla == "^":
            self.expresion += "**"
            self.entrada.set(self.expresion)

if __name__ == "__main__":
    root = tk.Tk()
    app = Calculadora(root)
    root.mainloop()
